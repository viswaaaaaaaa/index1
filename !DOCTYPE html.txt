  <script>
    document.addEventListener('DOMContentLoaded', function() {
      // Enhanced tab functionality for all tab containers
      document.querySelectorAll('.tabs-header').forEach(tabHeader => {
        const tabsContainer = tabHeader.closest('.tabs-container');
        
        tabHeader.addEventListener('click', function(e) {
          const btn = e.target.closest('.tab-btn');
          if (!btn) return;
          
          // Remove active class from all buttons in this header
          tabHeader.querySelectorAll('.tab-btn').forEach(b => {
            b.classList.remove('active');
          });
          
          // Remove active class from all contents in this container
          tabsContainer.querySelectorAll('.tab-content').forEach(content => {
            content.classList.remove('active');
          });
          
          // Add active class to clicked button
          btn.classList.add('active');
          
          // Show corresponding content
          const tabId = btn.getAttribute('data-tab');
          const targetContent = document.getElementById(tabId);
          if (targetContent) {
            targetContent.classList.add('active');
          }
        });
      });
    });
  </script>
